							import java.util.Scanner;

public class 부녀회장이될테야_2775 {
	public static void main(String[] args) {
		int[][] floor = new int[15][14];
		person(floor);
		Scanner sc = new Scanner(System.in);
		int t = sc.nextInt();
		for(int i=0; i < t; i++) {
			int k = sc.nextInt();
			int n = sc.nextInt();
			System.out.println(floor[k][n-1]);
		}
		sc.close();
	}

	public static void person(int[][] floor) {
		for(int i=0; i < 14; i++) {
			floor[0][i] = i+1;
		}
		for(int i=1; i < 15; i++) {
			floor[i][0] = 1;
		}
		for(int i=1; i < 15; i++) {
			for(int j=1; j < 14; j++) {
				floor[i][j] += floor[i-1][j]+floor[i][j-1];
			}
		}
	}
}
